
  Widget _buildPopFromCenter(BuildContext context) {
    return TDButton(
      content: '中间弹出',
      theme: TDButtonTheme.primary,
      type: TDButtonType.stroke,
      onTap: () {
        Navigator.of(context).push(TDSlidePopupRoute(
            slideTransitionFrom: SlideTransitionFrom.center,
            builder: (context) {
              return Container(
                color: Colors.white,
                width: 240,
                height: 240,
              );
            }));
      },
    );
  }
