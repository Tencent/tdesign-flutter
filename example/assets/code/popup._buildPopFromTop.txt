
  Widget _buildPopFromTop(BuildContext context) {
    return TDButton(
      text: '顶部弹出',
      theme: TDButtonTheme.primary,
      type: TDButtonType.outline,
      onTap: () {
        Navigator.of(context).push(TDSlidePopupRoute(
            slideTransitionFrom: SlideTransitionFrom.top,
            builder: (context) {
              return Container(
                color: Colors.white,
                height: 240,
              );
            }));
      },
    );
  }
