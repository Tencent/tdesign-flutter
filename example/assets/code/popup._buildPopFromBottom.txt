
  Widget _buildPopFromBottom(BuildContext context) {
    return TDButton(
      content: '底部弹出',
      theme: TDButtonTheme.primary,
      type: TDButtonType.stroke,
      onTap: () {
        Navigator.of(context).push(TDSlidePopupRoute(
            slideTransitionFrom: SlideTransitionFrom.bottom,
            builder: (context) {
              return Container(
                color: Colors.white,
                height: 240,
              );
            }));
      },
    );
  }
