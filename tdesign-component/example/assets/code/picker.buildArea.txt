
  Widget buildArea(BuildContext context) {
    const title = '选择地区';
    return TDCell(
        title: title,
        note: selected_1,
        arrow: true,
        onClick: (click) {
          TDPicker.showMultiPicker(context, title: title, onConfirm: (selected) {
            setState(() {
              selected_1 = '${data_1[selected[0]]}';
            });
            Navigator.of(context).pop();
          }, data: [data_1]);
        });
  }
