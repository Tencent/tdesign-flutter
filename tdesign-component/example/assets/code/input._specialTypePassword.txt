
  Widget _specialTypePassword(BuildContext context) {
    return Column(
      children: [
        TDInput(
          type: TDInputType.normal,
          controller: controller[12],
          obscureText: !browseOn,
          leftLabel: '输入密码',
          hintText: '请输入密码',
          rightBtn: browseOn
              ? Icon(
                  TDIcons.browse,
                  color: TDTheme.of(context).textColorPlaceholder,
                )
              : Icon(
                  TDIcons.browse_off,
                  color: TDTheme.of(context).textColorPlaceholder,
                ),
          onBtnTap: () {
            setState(() {
              browseOn = !browseOn;
            });
          },
          needClear: false,
        ),
        const SizedBox(
          height: 16,
        ),
      ],
    );
  }
