
  Widget _buildPopover(BuildContext context) {
    return Container(
      padding: const EdgeInsets.only(top: 0),
      child: LayoutBuilder(
        builder: (_, constraints) {
          return TDButton(
            text: '带箭头',
            type: TDButtonType.outline,
            theme: TDButtonTheme.primary,
            onTap: () {
              TDPopover.showPopover(context: _, content: '弹出气泡内容');
            },
          );
        },
      ),
    );
  }
