
  Widget _buildItemWithIcon(BuildContext context) {
    var tabs = List.generate(3, (index) {
      final text = '选项${index + 1}';
      return TDTab(
        text: text,
        icon: const Icon(TDIcons.app, size: 18),
      );
    });
    return TDTabBar(
      tabs: tabs,
      controller: TabController(length: tabs.length, vsync: this),
      showIndicator: true,
    );
  }
